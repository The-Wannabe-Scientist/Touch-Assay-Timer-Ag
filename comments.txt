3: * @description Web Worker that acts as an isolated metronome heartbeat.
4: *

4: //  Hardware: DRV2605L haptic driver (I²C) + ERM coin motor
5: //
6: //  ── WHAT THIS DOES ──────────────────────────────────────────
7: //  1. Dumps all key DRV2605L registers BEFORE and AFTER init.
8: //  2. Holds RTP output ON continuously for 10 s so you can
9: //     measure OUT+/OUT- with a multimeter without chasing pulses.
10: //  3. Prints STATUS register every second during the ON phase.
11: //
12: //  ── MOTOR TYPE ──────────────────────────────────────────────
13: //  Select your motor type by keeping ONE of the two lines below:
14: // #define MOTOR_LRA   // Linear Resonant Actuator (closed-loop, 1.8V RMS typical)
15: #define MOTOR_ERM   // Eccentric Rotating Mass coin motor (open-loop, 3V typical)
16: //
17: //  ── WIRING ──────────────────────────────────────────────────
18: //    DRV2605L SDA → D4 (XIAO I²C SDA)
19: //    DRV2605L SCL → D5 (XIAO I²C SCL)
20: //    DRV2605L VIN → Dual-voltage OR loop (5V USB + LiPo BAT+ directly via Schottky diodes)
21: //                   220µF–470µF decoupling cap across VIN/GND. Do NOT use XIAO 3.3V pin.
22: //    DRV2605L GND → GND
23: //    DRV2605L EN  → D3  (clone boards only; or hardwire EN to 3.3V)
24: //    Motor        → DRV2605L OUTP / OUTN
25: //                   (LRA: observe +/− polarity. ERM: polarity only affects spin direction)
26: //
27: //  ── EXTERNAL RGB LED (optional) ─────────────────────────────
28: //    R anode → 270Ω → D6
29: //    G anode → 100Ω → D7
30: //    B anode → 100Ω → D8
31: //    Cathode → GND
32: //    NOTE: Per-channel values. Green/blue Vf ≈ 2.8V — on 3.3V rail, 220Ω
33: //    leaves barely 2mA. 100Ω gives a proper 5mA brightness.
34: //    Green = init OK.  Red = DRV2605L not found.
35: // ============================================================

52: // Connect EN → D3 on the XIAO and the firmware drives it HIGH.
53: // If you hardwire EN to 3.3V instead, comment out this define.

74: // Common-cathode (default): HIGH = ON
83: pinMode(EXT_LED_R, OUTPUT); digitalWrite(EXT_LED_R, HIGH); // OFF
87: pinMode(EXT_LED_R, OUTPUT); digitalWrite(EXT_LED_R, LOW);  // OFF
97: // ── DRV2605L motor init (LRA) ────────────────────────────────
100: // Library 6 = LRA waveform library
103: // Set N_ERM_LRA bit (0x1A bit 7 = 1 → LRA mode)
106: // Enable closed-loop auto-resonance (0x1D bit 2 = LRA_OPEN_LOOP; clear = closed-loop)
109: // RATED_VOLTAGE: 1.8V RMS × 255 / 5.3 ≈ 87 = 0x57
110: // Adjust if your LRA has a different rated voltage.
113: // OD_CLAMP: 2.1V × 255 / 5.6 ≈ 96 = 0x60  (≈17% above rated)
116: // Auto-calibration — populates A_CAL_COMP (0x18) and A_CAL_BEMF (0x19)
136: // ── DRV2605L motor init (ERM) ────────────────────────────────
139: // Library 1 = ERM waveform library
142: // ERM mode helper (clears N_ERM_LRA bit 7 of 0x1A)
145: // Open-loop ERM: set ERM_OPEN_LOOP bit (0x1D bit 5 = 1)
146: // Recommended for coin ERMs — closed-loop back-EMF tracking causes 0V output
147: // on many small motors due to noisy or weak back-EMF signal.
150: // RATED_VOLTAGE: 3.0V × 255 / 5.3 ≈ 144 = 0x90
151: // Adjust for your specific ERM motor.
154: // OD_CLAMP: 3.3V × 255 / 5.6 ≈ 150 = 0x96
162: // ── Drive motor for onMs, silent for offMs, repeated reps times ─
166: drv.setRealtimeValue(127); // max amplitude (RTP signed 8-bit: 127 = full scale)
174: // ── Register dump helper ────────────────────────────────────
184: uint8_t feedBk  = drv.readRegister8(0x1A); // FEEDBACK_CTRL
185: uint8_t ctrl3   = drv.readRegister8(0x1D); // CONTROL3
195: // MODE: 0=INTTRIG 1=EXTTRIG_EDGE 2=EXTTRIG_LEVEL 3=PWM 4=AUDIO 5=RTP 6=DIAG 7=AUTOCAL
207: // 0=ERM, 1=LRA
211: // ERM_OPEN_LOOP bit5=1 → open-loop (required for small ERMs)
222: // ── EN pin HIGH before Wire.begin() ─────────────────────────
251: // ── ERM open-loop setup — explicit, step by step ─────────────
254: drv.selectLibrary(1);                    // Library 1 = ERM
255: drv.useERM();                            // clear N_ERM_LRA bit
258: c3 |= 0x20;                              // set ERM_OPEN_LOOP bit5
261: drv.writeRegister8(0x16, 0x90);          // RATED_VOLTAGE ~3.0 V
262: drv.writeRegister8(0x17, 0x96);          // OD_CLAMP ~3.3 V
264: // Explicitly clear STANDBY (bit6) and set MODE=0 (INTTRIG)
266: modeReg &= ~0x40;  // clear STANDBY
267: modeReg &= ~0x07;  // MODE = 0 (INTTRIG)
272: // ── Switch to RTP and hold ON for 10 s ───────────────────────
277: drv.setMode(DRV2605_MODE_REALTIME);  // 0x05 = RTP mode
278: drv.setRealtimeValue(127);           // 0x7F = max amplitude
